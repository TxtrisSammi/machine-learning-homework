\question $(b^{(\ell)},\p b^{(\ell)} / \p \mathcal{E})$ must have the same dimensions. Show the following pairs have the same dimensions,
\[
	\left(b^{(2)},\frac{\p \mathcal{E}}{\p b^{(2)}}\right), \quad \left(W^{(2)}, \frac{\p \mathcal{E}}{\p W^{(2)}}\right), \quad \left(b^{(1)}, \frac{\p \mathcal{E}}{\p b^{(1)}}\right), \quad \left(W^{(1)}, \frac{\p \mathcal{E}}{\p W^{(1)}}\right)
\]

\begin{sol}
	Because of the "rules for a three layered network" mentioned earlier, we said,

	\[
		b^{(\ell)} = b^{(\ell)} - \eta \left( \frac{\p \mathcal{E}}{\p b^{(\ell)}}\right)
	\]
	In order to do a matrix subtraction, both matrices must have the same dimensions. since $\eta$ is just a scalar value, we know that $b^{(\ell)}$ and $\left( \frac{\p \mathcal{E}}{\p b^{(\ell)}}\right)$ must have the same dimensions. The same applies for $W^{(\ell)}$ and $\left( \frac{\p \mathcal{E}}{\p W^{(\ell)}}\right)$ 
\end{sol}

\question Use the chain rule to expand the following derivative,
\[
	\frac{\text{d} f_n(f_{(n-1)}(f_{(n-1)}(\cdots f(x))))}{dx}
\]

\begin{sol}
\[
	= \frac{\text{d}f_n}{\text{d}f_{n-1}} \times \frac{\text{d}f_{n-1}}{\text{d}f_{n-2}} \times \cdots \times \frac{\text{d} f_1}{dx}
\]
\end{sol}
